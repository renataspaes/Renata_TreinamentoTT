int led = 2;
int button1 = 3;
int button2 = 4;
int inbutton1 = 0;
int inbutton2 = 0;


void setup()
{
  pinMode(button1, INPUT);
  pinMode(button2, INPUT);
  pinMode(led, OUTPUT);
}

void loop()
{
  inbutton1 = digitalRead(button1);
  inbutton2 = digitalRead(button2);
  // para a porta OR, apenas 0 e 0 dรก um resultado igual a 0
  if(inbutton1 == 0 && inbutton2 == 0){
  	digitalWrite(led, LOW);
  }
  else {
  	digitalWrite(led, HIGH);
  }
  delay(1000);
}
